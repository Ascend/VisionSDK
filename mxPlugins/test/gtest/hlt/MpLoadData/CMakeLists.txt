set(TARGET_EXECUTABLE "MpLoadDataTest")

set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${PROJECT_SOURCE_DIR}/dist/MpLoadData)

add_executable(${TARGET_EXECUTABLE} MpLoadDataTest.cpp)

target_link_libraries(${TARGET_EXECUTABLE} ${MXPLUGINS_HLT_TEST_COMMON_DEP_LIBS})

file(GLOB_RECURSE PIPELINE_FILES pipeline/*.pipeline)
file(GLOB_RECURSE INPUT_FILES input/*.json)
file(GLOB_RECURSE OUTPUT_FILES output/*.json)

install(FILES ${PIPELINE_FILES} DESTINATION ${CMAKE_RUNTIME_OUTPUT_DIRECTORY}/pipeline)
install(FILES ${INPUT_FILES} DESTINATION ${CMAKE_RUNTIME_OUTPUT_DIRECTORY}/input)
install(FILES ${OUTPUT_FILES} DESTINATION ${CMAKE_RUNTIME_OUTPUT_DIRECTORY}/output)

add_test(NAME ${TARGET_EXECUTABLE}
        COMMAND ${TARGET_EXECUTABLE} --gtest_output=xml
        WORKING_DIRECTORY ${CMAKE_RUNTIME_OUTPUT_DIRECTORY})
